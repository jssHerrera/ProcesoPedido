CREATE VIEW VIE_CLIENTES
AS
SELECT
   CLI.CLIENTE_ID,
   (CLI.RAZON_SOCIAL || '('|| CLI.NRO_DOCUMENTO ||')') NOMBRE,
   (PAI.SIGLA || ':'|| PAI.NOMBRE)PAIS
FROM
  CLIENTE CLI INNER JOIN PAIS PAI
  ON CLI.ID_PAIS=PAI.ID_PAIS
ORDER BY
   NOMBRE;

SELECT CLIENTE_ID,NOMBRE,PAIS FROM VIE_CLIENTES;

SELECT 
    CLIENTE_ID,
    NOMBRE,
    PAIS 
FROM 
    VIE_CLIENTES
WHERE
    (NOMBRE||PAIS) LIKE '%MEX%';
    
SELECT CLIENTE_ID,NOMBRE,PAIS FROM VIE_CLIENTES WHERE     UPPER(NOMBRE||PAIS) LIKE UPPER('%'||'CLA'||'%')
